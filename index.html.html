# Inventário automático — CSV + email
$unidades = Get-ADComputer -Filter * $relatorio = @()
foreach ($pc in $unidades) { try { $hw = Get-WmiObject Win32_ComputerSystem -CN $pc.Name $disk = Get-WmiObject Win32_LogicalDisk -CN $pc.Name $relatorio += [PSCustomObject]@{ Nome = $pc.Name RAM = "$([math]::Round($hw.TotalPhysicalMemory/1GB,1)) GB" DiskLivre = "$([math]::Round($disk.FreeSpace/1GB,1)) GB" Data = Get-Date -Format "yyyy-MM-dd" } } catch {} # offline — ignora }
$path = "inv_$(Get-Date -f yyyyMMdd).csv" $relatorio | Export-Csv $path -NoTypeInformation Send-MailMessage -To "gestao@pcpa.gov.br" -Attachments $path
# ✓ Executa toda segunda 07h00 — automático
